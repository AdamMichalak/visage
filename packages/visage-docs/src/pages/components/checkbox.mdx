import { Checkbox } from '@byteclaw/visage';
import { Fragment } from 'react';

# Checkbox

```jsx live=true
<Fragment>
  <Checkbox
    defaultChecked
    onChange={checked => console.log(checked)}
    label="Checkbox label"
  />
  <Checkbox hiddenLabel label="Checkbox label" />
  <Checkbox defaultChecked label="Uncontrolled checkbox" />
  <Checkbox checked label="Controlled checkbox" />
  <Checkbox label="Checkbox label" styles={{ fontSize: 2, lineHeight: 2 }} />
  <Checkbox disabled label="Disabled checkbox" />
  <Checkbox readOnly label="Read only checkbox" />
  <WithState>
    {(value, setValue) => (
      <Checkbox
        checked={!!value}
        onChange={checked => setValue(checked)}
        label="Controlled checkbox"
      />
    )}
  </WithState>
</Fragment>
```
