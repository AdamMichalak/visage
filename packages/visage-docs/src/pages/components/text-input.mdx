import { TextInput } from '@byteclaw/visage';

# Text input

TextInput component.

## Examples

### Default input

```jsx live=true
<TextInput placeholder="Placeholder text" />
```

### Default input with larger font size

```jsx live=true
<TextInput
  placeholder="Placeholder text"
  styles={{ fontSize: 1, lineHeight: 1 }}
/>
```

### Invalid input

```jsx live=true
<TextInput invalid placeholder="Placeholder text" />
```

### Disabled input

```jsx live=true
<TextInput
  disabled
  defaultValue="Disabled input"
  placeholder="Placeholder text"
/>
```

### ReadOnly input

```jsx live=true
<TextInput
  readOnly
  defaultValue="Disabled input"
  placeholder="Placeholder text"
/>
```

### Input with prepend

```jsx live=true
<TextInput prepend={<div>A</div>} placeholder="Placeholder text" />
```

### Input with append

```jsx live=true
<TextInput append="B" placeholder="Placeholder text" />
```

### Input with prepend and append

```jsx live=true
<TextInput append="B" prepend="A" placeholder="Placeholder text" />
```

### Support for refs

```jsx live=true
<WithRef>
  {ref => (
    <Fragment>
      <TextInput ref={ref} />
      <Button onClick={() => ref.current.focus()} type="button">
        Focus input
      </Button>
    </Fragment>
  )}
</WithRef>
```
