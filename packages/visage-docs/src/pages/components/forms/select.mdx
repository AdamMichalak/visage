# Select

## Import

```js transpile=false
import { Select } from '@byteclaw/visage';
```

## Examples

### Nonsearchable Select

```jsx live=true
<form action="">
  <label htmlFor="nonsearchable">Select nonsearchable value</label>
  <Select
    id="nonsearchable"
    onChange={value => console.log(value)}
    options={() => Promise.resolve(['a', 'b', 'c', 'd'])}
    placeholder="Select a value"
  />
</form>
```

### Searchable Select

```jsx live=true
<form action="">
  <label htmlFor="searchable">Searchable select value</label>
  <Select
    id="searchable"
    onChange={value => console.log(value)}
    options={() => Promise.resolve(['a', 'b', 'c', 'd'])}
    placeholder="Select a value"
    searchable
  />
</form>
```

### Disabled Select

```jsx live=true
<Select id="disabled-select" disabled placeholder="Select a value" />
```

### Invalid Select

```jsx live=true
<Select
  id="invalid-select"
  invalid
  searchable
  onChange={value => console.log(value)}
  options={filter =>
    new Promise(resolve =>
      setTimeout(
        resolve,
        1000,
        ['A', 'B', 'C', 'D', 'E', 'F'].filter(val =>
          !filter ? true : val === filter,
        ),
      ),
    )
  }
  placeholder="Select a value"
/>
```
