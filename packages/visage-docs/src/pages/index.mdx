```jsx transpile=false
import {
  createNPointTheme,
  ResponsiveDesignSystem,
  Text,
} from '@byteclaw/visage';

const theme = createNPointTheme();

<ResponsiveDesignSystem theme={theme}>
  <Text>Hello world!</Text>
</ResponseDesignSystem>
```
